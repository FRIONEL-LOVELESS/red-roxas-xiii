<script>
    window.onload = () => document.getElementById('output').innerText = "> [VIGILANCE]: SYSTEM VERIFIED // STATUS: OKAY.";

    async function executeCLI(cmd) {
        const output = document.getElementById('cli-output');
        const parts = cmd.split(' ');
        const token = document.getElementById('vault-key').value;
        
        switch(parts[0].toUpperCase()) {
            case 'PING':
                output.innerText = "> [SYSTEM]: PINGING REPOSITORY...";
                const check = await fetch('https://api.github.com/repos/FRIONEL-LOVELESS/red-roxas-xiii', {
                    headers: { 'Authorization': 'token ' + token }
                });
                output.innerText = check.ok ? "> [SYSTEM]: CONNECTION VERIFIED // ACCESS GRANTED." : "> [SYSTEM]: CONNECTION REJECTED // CHECK TOKEN SCOPE.";
                break;
            case 'FETCH':
                output.innerText = "> [SYSTEM]: DOWNLOADING README.md...";
                const res = await fetch('https://api.github.com/repos/FRIONEL-LOVELESS/red-roxas-xiii/contents/README.md', {
                    headers: { 'Authorization': 'token ' + token, 'Accept': 'application/vnd.github.v3.raw' }
                });
                const text = await res.text();
                output.innerText = "> [README_CONTENT]:\n" + text;
                break;
            case 'IMPORT': 
                document.getElementById('file-browser').click();
                output.innerText = "> [ABYSS_ADDER]: SELECT_FILE_TARGET...";
                break;
            case 'HYPER': 
                output.innerText = "> [HYPERTUNNEL]: ENGAGING...";
                await linkToGitHub('README.md', "THE ABYSS TERMINAL // HYPERTUNNEL\n\n- Chrono Trigger I: Identity Protocol (V3)\n- Chrono Trigger II: Hypertunnel Uplink\n- Chrono Trigger III: The Abyss Adder");
                break;
            case 'FORCE_HYPER':
                output.innerText = "> [SYSTEM]: FORCING HYPERTUNNEL...";
                const forceContent = "THE ABYSS TERMINAL // HYPERTUNNEL\n\n- Chrono Trigger I: Identity Protocol (V3)\n- Chrono Trigger II: Hypertunnel Uplink\n- Chrono Trigger III: The Abyss Adder";
                await linkToGitHub('README.md', forceContent);
                break;
            default: output.innerText = "> [ERROR]: COMMAND NOT RECOGNIZED.";
        }
    }

    function handleFile(files) {
        const reader = new FileReader();
        reader.onload = (e) => linkToGitHub(files[0].name, e.target.result);
        reader.readAsText(files[0]);
    }

    async function linkToGitHub(repoPath, content) {
        const token = document.getElementById('vault-key').value;
        const url = `https://api.github.com/repos/FRIONEL-LOVELESS/red-roxas-xiii/contents/${repoPath}`;
        try {
            const getRes = await fetch(url, { headers: { 'Authorization': 'token ' + token }});
            const fileData = await getRes.json();
            const payload = { message: "Hypertunnel Update", content: btoa(content), branch: "main" };
            if (fileData.sha) payload.sha = fileData.sha;
            await fetch(url, { method: 'PUT', headers: { 'Authorization': 'token ' + token, 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
            document.getElementById('cli-output').innerText = "> [ABYSS_ADDER]: SUCCESS // PATH: " + repoPath;
        } catch (err) {
            document.getElementById('cli-output').innerText = "> [ABYSS_ADDER]: FAILED // CHECK TOKEN/PERMISSIONS.";
        }
    }
